data AExp : Set

data AExp where
  $_ : (n : ℕ) → AExp
  `_ : (x : VarName) → AExp
  _+_ : (e₀ e₁ : AExp) → AExp
  _·_ : (e₀ e₁ : AExp) → AExp
  App : AExp → AExp → AExp
  Abs : (f : VarName → AExp) → AExp

infix 50 $_ `_

fv : AExp → VarName *
fresh : AExp → VarName

fv ($ n) = ε
fv (` x) = [ x ]
fv (e₀ + e₁) = fv e₀ ++ fv e₁
fv (e₀ · e₁) = fv e₀ ++ fv e₁
fv (App e₀ e₁) = fv e₀ ++ fv e₁
fv (Abs f) = {!   !}
